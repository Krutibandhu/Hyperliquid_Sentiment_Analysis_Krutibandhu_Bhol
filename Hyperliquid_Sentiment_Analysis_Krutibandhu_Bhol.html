fee_analysis = merged_df.groupby('classification').agg(
    total_pnl=('Closed PnL', 'sum'),
    total_fees=('Fee', 'sum'),

).reset_index()

# Net PnL after subtracting transaction fees
fee_analysis['net_pnl'] = fee_analysis['total_pnl'] - fee_analysis['total_fees']
